# [LeetCode 5] 反转原字符串后使用滑动窗口

毫无疑问，这一题的“标准”解法是dp。但回文串的性质是什么？反转前后相等。所以，可以利用这个性质，寻找反转后的字符串与原字符串中最长的相同部分，应该就是我们要寻找的回文子串。这个过程可以用滑动窗口来实现。

但是这一点并不完全正确，比如，对于"aacdefcaa"这个字符串来说，反转后是"aacfedcaa"，反转前后最长的相同部分是"aac"，但这并不是一个回文串。所以，我们还需要对反转前后的字符串进行一次判断，如果是回文串才加入答案。

function longestPalindrome(s: string): string {
  function isPalindrome(s: string) {
    let b = 0,
      e = s.length - 1;
    while (b < e && s[b] === s[e]) {
      ++b, --e;
    }
    return b >= e;
  }

  const s2 = s.split("").reverse().join(""),
    m = s.length,
    n = s2.length;
  let res = "";
  // slide s2 first
  for (let i = 0; i < m; ++i) {
    const window = Math.min(n, m - i);
    let b = i,
      max = 0;
    for (let j = 0; j < window; ++j) {
      if (s[i + j] === s2[j]) ++max;
      else {
        b = i + j + 1;
        max = 0;
      }
      // console.log(b, max, res);
      if (max > res.length) {
        const tmp = s.substr(b, max);
        if (isPalindrome(tmp)) res = tmp;
      }
    }
  }
  // slide s then
  for (let i = 0; i < n; ++i) {
    const window = Math.min(m, n - i);
    let b = i,
      max = 0;
    for (let j = 0; j < window; ++j) {
      if (s2[i + j] === s[j]) ++max;
      else {
        b = i + j + 1;
        max = 0;
      }
      if (max > res.length) {
        const tmp = s2.substr(b, max);
        if (isPalindrome(tmp)) res = tmp;
      }
    }
  }
  return res;
}
